Data mapping


              query {
                  posts {
                    title
                    body
                  }
              }
          

              schema {
                query: Query
              }
            
              type Query {
                posts: [Post]!
              }

Data mapping


              const resolvers = {
                Query: {
                  posts: () => fetch('/v1/posts')
                }
              };
          

              type Query {
                posts: [Post]!
              }

              type Post {
                title: String!
                body: String!
              }

Data mapping


                const resolvers = {
                  Post: {
                    title: post => post.title, // Default implementation
                    body: post => post.body
                  }
                };
            

                type Post {
                  title: String!
                  body: String!
                }

Data mapping


                query {
                    posts {
                      title
                      author {
                        name
                      }
                    }
                }

Data mapping


                const resolvers = {
                  ...
                  Post: {
                    author: post =>
                      fetch('/v1/author/' + post.authorId)
                  },
                };
            

                type Query {
                  posts: [Post]!
                }
  
                type Post {
                  title: String!
                  author: Author
                }

                type Author {
                  name: String!
                }

Data mapping


                  query {
                      posts {
                        title
                        author {
                          name
                        }
                      }
                  }

Data mapping


                const resolvers = {
                  Query: {
                    posts: (_, { limit }) => 
                      fetch('/v1/posts').then(posts => posts.slice(0, limit))
                  }
                };
              

                  type Query {
                    posts(limit: Int): [Post]!
                  }
